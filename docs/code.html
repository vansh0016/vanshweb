library(ggplot2)     # Loads ggplot2
library(tidyverse)   # Loads tidyverse
library(plotly)      # Loads plotly
library(scales)      # Loads scales

world_population <- read.csv("world_population.csv")  # Reads the data into a data frame

world_population <- world_population %>%
  rename(country = `Country.Territory`,  # Renames 'Country.Territory' to 'country'
         capital = `Capital`,              # Renames 'Capital' to 'capital'
         population_2020 = `X2020.Population`)  # Renames 'X2020.Population' to 'population_2020'

plot <- plot_ly(
  data = world_population,  # Uses the cleaned world_population dataset
  type = 'choropleth',      # Specifies the type of plot as a choropleth map
  locations = world_population$CCA3,  # Assigns country codes (CCA3) for geographical locations
  z = world_population$population_2020,  # Sets the population values for color intensity
  text = paste("Country:", world_population$country,  # Creates hover text to show detailed information
               "<br>Capital:", world_population$capital,
               "<br>Population:", world_population$population_2020,
               "<br>Population Percentage:", world_population$World.Population.Percentage),
  hoverinfo = "text",  # Displays the detailed hover text when the user hovers over a country
  colorscale = list(c(0, "lightblue"), c(1, "darkblue")),  # Defines a color scale from light blue to dark blue
  colorbar = list(title = "Population"),  # Labels the color bar indicating the population scale
  marker = list(line = list(color = 'black', width = 0.5))  # Sets the border color and width for each country
) %>%
layout(
    title = 'World Population',  # Sets the main title of the plot
    geo = list(
      showframe = FALSE,  # Hides the frame around the map for a cleaner look
      showcoastlines = TRUE,  # Enables coastlines to help delineate country borders
      coastlinecolor = 'Black',  # Sets the color of the coastlines to black
      projection = list(type = 'Mercator'),  # Uses Mercator projection for accurate representation of landmasses
      showland = TRUE,  # Ensures that land areas are displayed on the map
      landcolor = 'lightgrey',  # Sets the color of land areas to light grey for contrast
      subunitcolor = 'black'  # Sets the color of subunit borders (countries) to black for visibility
    )
)

plot  # Displays the plot
Code for Redesign 2
world_population2 <- read.csv("world_population.csv")  # Loads the dataset

long_population_data <- world_population2 %>%
  select(`Country.Territory`,  # Selects the country territory column
         `X2022.Population`,   # Selects the population for 2022
         `X2020.Population`,   # Selects the population for 2020
         `X2015.Population`,   # Selects the population for 2015
         `X2010.Population`,   # Selects the population for 2010
         `X2000.Population`,   # Selects the population for 2000
         `X1990.Population`,   # Selects the population for 1990
         `X1980.Population`,   # Selects the population for 1980
         `X1970.Population`) %>%  # Selects the population for 1970
  pivot_longer(cols = starts_with("X"),  # Reshapes the data to long format for years
               names_to = "Year",  # Sets the name for the year column
               values_to = "Population") %>%  # Sets the name for the population column
  mutate(Year = gsub("X", "", Year),  # Removes "X" from the year values
         Year = gsub("\\.Population", "", Year))  # Removes ".Population" from the year values

top_ten_countries <- world_population2 %>%
  arrange(desc(`X2020.Population`)) %>%  # Arranges the countries in descending order by 2020 population
  head(10) %>%  # Selects the top ten countries
  pull(`Country.Territory`)  # Extracts the country names

filtered_data <- long_population_data %>%
  filter(`Country.Territory` %in% top_ten_countries)  # Filters data for the top ten countries

plots_list <- list()  # Creates an empty list to hold plot objects

for (country in unique(filtered_data$`Country.Territory`)) {
  country_data <- filtered_data %>% filter(`Country.Territory` == country)  # Filter data for the current country

  plot <- plot_ly(data = country_data,
                  x = ~Year,  # Sets the x-axis to the Year
                  y = ~Population,  # Sets the y-axis to Population
                  type = 'scatter',  # Specifies the plot type as scatter
                  mode = 'lines+markers',  # Uses lines and markers for the plot
                  name = country,  # Labels the plot with the country's name
                  text = ~paste("Population:", Population),  # Creates hover text showing population
                  hoverinfo = 'text') %>%  # Ensures hover info displays the text
    layout(title = country,  # Sets the title of the plot to the country's name
           xaxis = list(title = "Year"),  # Labels the x-axis
           yaxis = list(title = "Population"),  # Labels the y-axis
           showlegend = FALSE)  # Disables the legend for individual country plots

  plots_list[[country]] <- plot  # Adds the plot to the list using the country name as the key
}

combined_plot <- subplot(plots_list, nrows = 2, shareX = TRUE, titleX = TRUE, titleY = TRUE) %>%
  layout(title = "Change in Population of Top 10 Countries Over Years",  # Sets the main title for the combined plot
         xaxis = list(title = "Year"),  # Labels the x-axis for the combined plot
         yaxis = list(title = "Population"),  # Labels the y-axis for the combined plot
         showlegend = TRUE)  # Enables the legend for the combined plot

combined_plot  # Displays the final combined plot of population changes

library(ggplot2)  # Loads ggplot2
library(plotly)   # Loads plotly
library(dplyr)    # Loads dplyr

Book1 <- read.csv("Book1.csv")  # Loads the data into a data frame

Book1 <- Book1 %>%
  rename(
    State = State,  # Renames State column to State
    Gun.Ownership.Rate = `Gun.Ownership.Rate....`,  # Renames gun ownership rate column
    Suicidal.Rate.Per.100000 = `Suicidal.rate..per.100.000.`  # Renames suicidal rate column
  )

scatter_plot <- ggplot(Book1, aes(x = Gun.Ownership.Rate, y = Suicidal.Rate.Per.100000, color = State)) +
  geom_point(size = 3, alpha = 0.7) +  # Adds points for each state with specified size and transparency
  geom_smooth(method = "lm", se = FALSE, color = "black", linetype = "dashed") +  # Adds a dashed regression line without confidence intervals
  labs(title = "Gun Ownership vs Suicidal Rate by State",  # Sets the main title of the plot
       x = "Gun Ownership Rate (%)",  # Labels the x-axis
       y = "Suicidal Rate") +  # Labels the y-axis
  theme(legend.position = "none")  # Hides the legend

interactive_plot <- ggplotly(scatter_plot)  # Creates an interactive version of the scatter plot

interactive_plot  # Renders the interactive plot for user interaction

library(reshape2)  # Loads reshape2
library(ggplot2)   # Loads ggplot2

Book1 <- read.csv("Book1.csv")  # Loads the data into a data frame

Book1 <- Book1 %>%
  rename(
    State = State,  # Renames State column to State
    Gun.Ownership.Rate = `Gun.Ownership.Rate....`,  # Renames gun ownership rate column
    Suicidal.Rate.Per.100000 = `Suicidal.rate..per.100.000.`  # Renames suicidal rate column
  )

df_long <- melt(Book1, id.vars = "State", measure.vars = c("Suicidal.Rate.Per.100000", "Gun.Ownership.Rate"))  # Melts the data frame

colnames(df_long) <- c("State", "Metric", "Value")  # Updates column names

df_long$Metric <- factor(df_long$Metric, levels = c("Gun.Ownership.Rate", "Suicidal.Rate.Per.100000"))  # Sets order for plotting

total_values <- df_long %>%
  group_by(State) %>%
  summarise(Total = sum(Value, na.rm = TRUE))  # Summarizes total values by state, ignoring NAs

df_long <- df_long %>%
  left_join(total_values, by = "State") %>%
  arrange(Total)  # Arranges data by total value in ascending order

ggplot(df_long, aes(x = reorder(State, Total), y = Value, fill = Metric)) +  # Initializes ggplot with specified aesthetics
  geom_bar(stat = "identity", na.rm = TRUE) +  # Creates bars using the identity statistic, ignoring missing values
  geom_text(aes(label = round(Value, 1)), position = position_stack(vjust = 0.5), size = 2, na.rm = TRUE) +  # Adds text labels inside the bars
  scale_fill_manual(values = c("Suicidal.Rate.Per.100000" = "red", "Gun.Ownership.Rate" = "skyblue"),  # Defines colors for the metrics
                    labels = c("Suicidal.Rate.Per.100000" = "Suicidal Rate",  # Custom legend labels
                               "Gun.Ownership.Rate" = "Gun Ownership Rate")) +
  labs(title = "Stacked Bar Plot of Gun Ownership Rate and Suicidal Rate by State",  # Sets the plot title
       x = "State",  # Labels the x-axis
       y = "Rate",  # Labels the y-axis
       fill = "Metric") +  # Labels the fill legend
  theme(axis.text.x = element_text(angle = 0, size = 8)) +  # Customizes x-axis text appearance
  coord_flip()  # Flips the coordinates for better readability

library(ggplot2)  # Loads ggplot2
library(tidyverse)  # Loads tidyverse

debatestats <- read.csv("PDebate.csv")  # Loads the data into a data frame

long_debatestats <- debatestats %>%
  pivot_longer(cols = c(Trump, Biden),  # Specifies the columns to pivot
               names_to = "candidate",  # New column to hold candidate names
               values_to = "score")  # New column to hold the scores (time in minutes)

 ggplot(long_debatestats, aes(x = Topic, y = score, fill = candidate)) +  # Initializes ggplot with specified aesthetics
  geom_bar(stat = "identity", position = "stack") +  # Creates stacked bars using the identity statistic
  scale_fill_manual(values = c("Biden" = "blue", "Trump" = "red")) +  # Defines colors for each candidate
  geom_text(aes(label = score), position = position_stack(vjust = 0.5), color = "white") +  # Adds text labels showing the score inside the bars
  labs(title = "Debate Stats: Trump vs Biden by Topic",  # Sets the plot title
       x = "Topic",  # Labels the x-axis
       y = "Time (mins)",  # Labels the y-axis indicating time in minutes
       fill = "Candidate")  # Labels the fill legend for candidate colors
